// Тема 17 - Массивы
#include <iostream>

using namespace std;

int main()
{
    // Массив это упорядоченный набор однотипных переменных. 
    // Каждая переменная в массиве является самостоятельной единицей, и называется его элементом
    // Каждый элемент массива имеет свой номер(индекс)
    // Индексы элементов массива начинаются с нуля
    
    // Объявление массива
    int arr[5] = {2,4,5,7,8};
    int arr2[] {1,2,3}; // Если сразу инициализировать массив - необязательно указывать его размер
    int arr3[4] = {}; // Неинизиализированные элементы массива, по умолчанию = 0
    
    cout << arr[1] << endl;  // Обращение к элементу по индексу
    cout << arr2[1] << endl;  
    cout << arr3[1] << endl;  // 0
    
    // Ввод массива с клавиатуры
    for(int i = 0; i < 4; i++){
        cin >> arr3[i];
    }
    cout << arr3[1] << endl;
}
