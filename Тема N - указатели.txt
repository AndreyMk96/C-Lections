//указатель - это переменная, которая содержит адрес другой переменной или объекта

int main()
{
	int a = 5;
	int *px = &a;
	//double *px = &a;//так нельзя, тип данных не подходит
	cout << px << endl;//вывелся адрес в памяти
	cout << *px << endl;//выведется значение
	int *px2 = &a;
	cout << px2 << endl;//адрес тот же самый
	*px2 = 10;//через указатель меняется значение переменной
	cout << a << endl;
	cout << *px << endl;//у всех указателей на эту переменную меняется значение
	_getch();
    return 0;
}
