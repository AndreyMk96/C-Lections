/*Дело в том что в C++ в функцию можно передать только указатель (или ссылку) на массив, соответственно вы всегда будете работать 
с исходным массивом, а не с его копией (и вернете также указатель на исходный массив)*/

int *funk(int arr[10]){
	for (int i = 0; i < 5; i++){
		arr[i] = arr[i] * 2;
	}
	return arr;
}
int main()
{
	int arr[10] = { 1, 3, 2, 4, 5 };
	cout << arr[0];
	funk(arr);
	cout << arr[0];
	_getch();
	return 0;
}
