// Тема 10 - Цикл For

#include <iostream>

using namespace std;

int main()
{
    // В цикле for указываются три параметра
    // 1 - Какая переменная инициализируется итератором цикла
    // 2 - Проверка условия выполнения цикла
    // 3 - Шаг изменения итератора
    for (int i = 0; i < 10; i++){
        cout<<"Hello World" << endl;
    }
    
    int i = 0;  // Переменная может быть инициализирована вне цикла
    for ( ; i < 10; i+=2){ // Тогда первый параметр может быть пропущен
        cout<< i << endl;
    }
    // Это может использоваться, если у нас несколько циклов
    for ( ; i < 20; ){ // Тогда первый параметр может быть пропущен
        cout<< i << endl;
        i+=4; // Третий параметр может быть вынесен внутрь цикла
    }   
    
    // Если переменная объявлена внутри цикла, то она не может использоваться снаружи
    for (int j = 0; j < 1; j++){
        cout << endl;
    }
    //cout << j; error
    // В цикла For также может быть несколько переменных
    for(int k = 0,  l = 10; k == l; k++, l--){
        cout << k << " " << l << endl;
    }
    
    // Если в цикле отсутствует проверка условия выполнения из цикла, то он получится бесконечным
    for (; ; ){
        cout << "Цикл выполняется" << endl;
    }

    return 0;
}
