int main()
{
	int a = 1;
	//постфиксная форма
	a++;
	cout << a << endl;//2
	a--;
	cout << a << endl;//1
	//префиксная форма
	++a;
	cout << a << endl;//2
	--a;
	cout << a << endl;//1
	cout << a++ << endl;//1, так как приоритет постфиксного инкремента ниже чем у остальных операций
	cout << ++a << endl;//3 так как приоритет префиксного инкремента выше, чем у остальных операций
	int b = ++a * a;
	cout << b;//16
	_getch();
    return 0;
}
