// Тема N - Динамические массивы
#include <iostream>
#include <cstdlib>

using namespace std;

int main()
{
    // Под динамические массивы место выделяется в динамической памяти
    // Размер динамического массива можно установить по ходу выполнения программы

    int size = 5;
    int *arr = new int[size]{1,3,5,7,9}; // Создаем указатель на динамический массив

    // Вывод динамического массива через указатель
    for(int i = 0; i < size; i++){
        cout << *(arr + i) << " ";
    }
    cout << "\n";

    // Генерация динамического массива из случайных чисел
    int size2;
    cout << "Введите размер массива"<< endl;
    cin >> size2;
    int *arr2 = new int[size2];
    for (int i =0; i < size2; i++){
        *(arr2 + i) = rand() % 10;
    }
    for(int i = 0; i < size2; i++){
        cout << *(arr2 + i) << " ";
    }
    delete [] arr;  // При очистке дин. массива ставятся квадратные скобки после delete
    delete [] arr2;
}
