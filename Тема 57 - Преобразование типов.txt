// Тема 58 - Преобразование типов

#include <iostream>

using namespace std;

int main()
{
    double a = 100.5;
    cout << (int)a << endl; // Преобразование в int(не по правилам математики, а отбросом части после запятой)
    cout << (bool)a << endl;  // Преобразование в bool
    cout << (char)a << endl;  // Преобразовалось в символ d по таблице ASCII
    
    int b = (int)a;  // Приведение к другом типу путем занесения в другую переменную
    cout << b << endl;
    // Такое преобразование(где конкретно указывается тип назыается явным)
    // Когда копилятор автоматически преобразует переменную в другой тип, такое преобразование называется неявным
    int c = 10.123123;  // Неявно преобразуется в int(сужающее преобразование)
    // C неявным преобразованием надо быть осторожным. Есть риск потери данных
    double d = 10;
    cout << d << endl; // Расширяющее преобразоавание
    cout << 11.5/44 << endl; // Сужающее преобразование
    return 0;
}
