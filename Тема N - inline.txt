//inline используют для оптимизации быстродействия кода
//позволяет не тратить время компилятору на вызов функции
//подходит для маленьких функций, которые можно "встроить" в код
//в современных компиляторах не актуально
inline int Sum(int a, int b) {
	return a + b;
}

int main()
{
	cout << Sum(5, 10);
	_getch();
    return 0;
}
