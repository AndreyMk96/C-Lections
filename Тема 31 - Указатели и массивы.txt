using namespace std;
//указатели и массивы. Арифметика указателей


int main()
{
	const int SIZE = 5;
	int arr[SIZE] = { 1,3,5,7,9 };
	for (int i = 0; i < 5; i++) {
		cout << arr[i] << endl;
	}
	//имя массива - это указатель на его первый элемент
	int *pArr = arr;//присвоили указателю ссылку на массив
	for (int i = 0; i < 5; i++) {
		cout << pArr[i] << endl;
	}
	//одинаковые адреса в памяти
	cout << arr << endl;
	cout << pArr << endl;
	//вывести элемент через указатель,
	//не забываем, что индексы начинаются с нуля
	cout << *(pArr + 3) << endl;
	//получится тоже самое, так как название массива,
	//это указатель на его нулевой элемент
	cout << *(arr + 3) << endl;
	cout << *(arr + 100) << endl;//ошибка

	_getch();
    return 0;
}
