// Тема N - Рекурсия

#include <iostream>

using namespace std;

int func(int a){
    if (a < 0)  // В рекурсивной функции должно быть условие выхода из нее
        return 0;
    cout << a << " " << &a << endl; // Также видно, что каждая функция занимает свое место в памяти
    return func(a - 1);
}

int main()
{
    //Рекурсия - это вызов функцией самой себя
    func(10);  // Пример вызова рекурсивной функции

    return 0;
}
