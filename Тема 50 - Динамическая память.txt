// Тема N - Динамическая память
#include <iostream>

using namespace std;

int main()
{
    // Динамическая память - это область памяти, не занятая операционной системой 
    // или другими загруженными в данный момент программами.
    // Другими словами это память - взятая программой извне(из оперативной памяти)
    
    // Обращение к динамической памяти происходит через указатели
    
    // Выделение динамической памяти происходит через оператор new
    int * pointer_a = new int;  // Создаем указатель на участок динамической памяти
    cout << pointer_a << endl;  // Выводитя ссылка на участок динамической памяти
    *pointer_a = 10;  // Присвоим значение через оператор разыменования 
    cout << *pointer_a << endl;
    
    // В С++ нет автоматического сборщика мусора. Объекты надо удалять из памяти вручную
    delete pointer_a;  // Необходимо удалять объекты чтобы не было утечки памяти
}
