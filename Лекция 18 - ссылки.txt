using namespace std;
//ссылки - это переменная, которая как и указатель хранит адрес, где лежат какие-то данные

int main()
{
    int a = 5;
    int *pa = &a;//это указатель
    int &aRef = a;//а это ссылка
    cout << pa << " " << aRef << endl;//по указателю выводится адрес в памяти. а по ссылке - значение
    //ссылки в отличии от указателей работают сразу со значением, на которое они указывают
    //* в указателе - это оператор разименования, со звездочкой выводится значение
    pa++;
    cout << pa << endl;//видим что адрес изменился на 4 байта
    aRef++;
    cout << aRef << endl;//при работе по ссылке менятется значение
    aRef = 55;
    cout << a<<endl;//по ссылке можно изменить значение переменной
    int* p;//указателю можно не присваивать значение
    //int& pRef; //ссылка должна быть инициализирована
    int* ppa = &aRef;//указателю можно присвоить ссылку
    cout << ppa << endl;//адрес
    cout << *ppa << endl;//значение
    *ppa = 10;
    cout << a;

}
