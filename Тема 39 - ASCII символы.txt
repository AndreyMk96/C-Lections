// Тема 62 - ASCII - символы

#include <iostream>

using namespace std;

int main()
{
    setlocale(LC_ALL, "ru");
    cout<<sizeof(char)<< endl; // Тип char занимает 1 байт(8 бит)
    // 8 бит может хранить 256 символов (2 в 8 степени)
    // по сути тип char хранит числа от 0 до 255

    for(int i = 0; i < 255; i++){
        cout << char(i) << " "; // Приводим int в char
    }
    // часть символов могут не отображаться
    // Символы 0-127 всегда одинаковые(это английская раскладка клавиатуы)
    // Символы 128 - 255 - это setlocale(в нашел случае ru)
    return 0;
}
