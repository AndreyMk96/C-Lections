// Тема N - Функции и методы строк (изменение строк)

#include <iostream>

using namespace std;

int main()
{
    string str = "Hi, my dear Friend";
    
    //Если надо добавить в конец строки другую строку, применяется метод append(), в который передается добавляемая строка
    str.append("!");
    cout << str << endl;
    
    // Для вставки одной строки в другую применяется функция insert()
    str.insert(10, "*");
    cout << str << endl;
    // Задача. Вставить символ * после каждой гласной буквы в строке
    
    // Для замены в строке некоторой части применяется функция replace()
    str.replace(8, 3, "*");  // Заменяем три символа начиная с 8 на *
    cout << str << endl;
    // Задача, заменить все вхождения в строке состоящие из 2-х и более гласных подряд на символ *
    // учитывайте, что длина строки меняется
    
    // Если надо не просто заменить символы, а удалить их из текста, также можно использовать функцию replace() 
    // - в этом случае удаляемые символы фактически заменяются на пустую строку
    str.replace(5, 2, "");  // Удалили два символа начиная с пятого
    cout << str << endl;
    // Однако С++ также предоставляет для удаления символов специальную функцию - erase(). 
    // В качестве параметров она принимает начальный индекс удаления и количество удаляемых символов:
    str.erase(5, 2); // Удалили еще 2 символа начиная с пятого
    cout << str << endl;
    // Задача. Удалить из текста все вхождения из 2-х подряд идущих согласных букв
    
    // Если в erase() передан один параметр, то удалится вся строка начиная от этого индекса
    str.erase(10);
    cout << str << endl;
    
    // Стоит отметить, что в стандарт С++20 была добавлена функция std::erase(), 
    // которая удаляет все вхождения определенного символа в строке (работает не во всех компилсяторах)
    std::erase(str, 'i');
    cout << str << endl;
    
    return 0;
}
