// Тема 47 - Запись текстовых файлов
#include <iostream>
#include <fstream>

using namespace std;

int main()
{
    // Для записи в файл к объекту ofstream или fstream применяется оператор << (как и при выводе на консоль):
    std::ofstream out;          // поток для записи
    out.open("test_write.txt");      // открываем файл для записи
    if (out.is_open())
    {
        out << "Hello World!" << std::endl;
    }
    out.close();
    // Если текстового документа с таким именем нет, он создастся автоматически
    // Если с таким именем есть не пустой файл, он перезапишется

    // Дозапись файла
    std::ofstream out2;
    out2.open("test_write.txt", std::ios::app);
    if (out2.is_open())
    {
        out2 << "Hello World! 2" << std::endl;
    }
    out.close();
    // Задачи
    // 1) Даны два текстовых файла. Поменять их содержимое местами
    // 2) Даны текстовый файл. Записать его содержимое в другой файл в обратном порядке
    // 3) Имеется текстовый файл prices.txt с информацией о заказе из интернет магазина.
    // В нем каждая строка с помощью символа табуляции \t разделена на три колонки:
    // наименование товара;
    // количество товара (целое число);
    // цена (в рублях) товара за 1 шт. (целое число).
    // Напишите программу, подсчитывающую общую стоимость заказа.
}
