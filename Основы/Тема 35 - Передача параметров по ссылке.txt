using namespace std;
//ссылочные параметры. Передача аргументов по ссылке
void Foo(int a) {
	a = 1;
}

void Foo2(int &a) {//функция принимает ссылку на значение пеерменной
	a = 2;//через адрес(ссылку) напрямую к данным присваивается значение
}

void Foo3(int *a) {//функция принимает указатель
	*a = 3;
}

void Foo4(int &a, int &b, int &c) {
	a = 10;
	b *= 2;
	c -= 100;
}


int main()
{
	int value = 5;
	Foo(value);
	cout << value<<endl;
	Foo2(value);
	cout << value << endl;
	Foo3(&value);//в данном случае передаем адрес переменной
	cout << value << endl;
	//при передаче ссылки в функцию она ведет себя также как указатель, только нам не нужно брать адрес при передаче параметра и нам не нужно выполнять 
	//операцию разименования при работе с данными на которые ссылается ссылка
	//ссылки можно использовать для возврата нескольких значений, как и указатели
	int a = 0, b = 10, c = 30;
	Foo4(a, b, c);
	cout << a << " " << b << " " << c << endl;
	//ссылочные типы работают с адресами в памяти
	//ДЗ написать ШАБЛОННУЮ функцию которая меняет местами значения 2 - х переменных с помощью ССЫЛОК.

 	_getch();
    return 0;
}
