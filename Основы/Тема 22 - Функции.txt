// Тема 21 - Функции

#include <iostream>

using namespace std;

// Функция типа void
void foo(){
    cout << "Вызвана функция" << endl;
}

// Функция типа int 
int sum(int a, int b){
    int result = a + b;
    return result; 
}

int main()
{
    // Функция это отдельно вынесенный участок кода, который принимает аргументы, и возвращает занчения
    // Перед объявлением функции, указывается тип ее возвращаемого значения, или тип void, если ф - ция ничего не возвращает
    
    // Параметры(аргументы) функции - это переменные, которые функция принимает
    // Для возврата значений из функции используется ключевое слово return
    
    foo(); // Вызов функции
    foo(); // Функцию можно вызывать сколько угодно раз
    
    int sm = sum(5, 10); // Если функция возвращает значение, его нужно принимать в переменную
    cout << sm << endl;
    // Функцию нельзя вызвать до ее объявления
    
    // Переменные могут передаваться в функцию по значению или по ссылке
    // При передаче по значению переменная копируется в функцию
    return 0;
}
