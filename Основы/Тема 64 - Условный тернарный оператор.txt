// Тема N - Условный тернарный оператор

#include <iostream>

using namespace std;

int main()
{
    // Как изучали ранее, операторы бывают унарные, бинарные, и тернарные
    // Тернарный оператор взаимодействует с тремя операндам
    // Запись ?: аналогичен конструкии if-else
    int a = 15;
    // Если условие будет проверяться тернарным оператором, мы должны поставить знак вопроса
    // Далее указываем действие, если значение истинно
    // В виде else выступает двоеточие
    // Выражение перез знаком ? проверяется на истинность
    a > 10 ? cout << "a > 10" << endl : cout << "a < 10" << endl;
    
    // В тернарных операторах можно выполнять любые действия, в т. ч. и изменять переменные
    a > 10 ? a -= 5 : a += 5;
    cout << a << endl;
    
    // Если в тернарном операторе необходимо выполнить более 2-х проверок,
    // То внутрь тернарного оператора, можно поместить еще один тернарный опреатор и т. д.
    int b = 11;
    b > 10 ? cout << "b > 10" << endl : b < 10 ? cout << "b < 10" << endl : cout << "b == 10" << endl;

    return 0;
}
