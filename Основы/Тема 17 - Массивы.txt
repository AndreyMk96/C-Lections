// Тема 17 - Массивы
#include <iostream>

using namespace std;

int main()
{
    // Массив это упорядоченный набор однотипных переменных. 
    // Каждая переменная в массиве является самостоятельной единицей, и называется его элементом
    // Каждый элемент массива имеет свой номер(индекс)
    // Индексы элементов массива начинаются с нуля
    
    // Объявление массива
    int arr[5] = {2,4,5,7,8};
    int arr2[] {1,2,3}; // Если сразу инициализировать массив - необязательно указывать его размер
    
    const int size = 3; // размер массива можно задавать константной переменной. Удобно при изменении кол - ва элементов в массиве
    int arr3[size] = {}; // Неинизиализированные элементы массива, по умолчанию = 0
    
    cout << arr[1] << endl;  // Обращение к элементу по индексу
    cout << arr2[1] << endl;  
    cout << arr3[1] << endl;  // 0
    
    cout << "Введите массив из 3 - элементов" << endl;
    // Ввод массива с клавиатуры
    for(int i = 0; i < size; i++){
        cin >> arr3[i];
    }
    cout << "Введенный массив" << endl;
    // Вывод массива
    for (int i = 0; i < size; i++){
        cout << arr3[i] << " ";
    }
}
