// Тема N - Математические операторы
#include <iostream>

using namespace std;

int main()
{
    // Операторы бывают
    // Унарные
    // Бинарные
    // Тернарные
    
    // Унарные - это операторы для действий которых нужен только один операнд
    // Бинарные - это операторы для действий которых нужно два операнда 
    // Тернарные - это операторы для действий которых нужно три операнда 
    int a = 10;
    a = -a;  // Унарный минус
    cout << a;
    
    int b = 10;
    
    cout << a + b << endl; // Сложение
    cout << a - b << endl; // Вычитание
    cout << a * b << endl; // Умножение
    cout << a / b << endl; // Деление
    cout << a % b << endl; // Остаток от деления (Деление по модулю)
    
    
    // Приоритет выполнения операций
    // Унарный минус
    // Действия в скобках
    // Умножение, деление, деление по модулю, остаток от деления
    // Сложение, вычитание
    
    a += 10; // Сокращенная форма записи, a = a + 10
    a -= 20  // Вычитание
    a *= 2;  // Умножение
    a /= 5;  // Деление
    a %= 10; // Остаток от деления
    
    // Сокращенные арифметические формы записи являются хорошим тоном, и улучшают читабельность кода
    return 0;
}
