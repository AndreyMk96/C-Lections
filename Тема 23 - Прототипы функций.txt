// Тема 24 - Прототипы функций

#include <iostream>

using namespace std;

// Прототипом функции в языке Си или C++ называется объявление функции, не содержащее тела функции, 
// но указывающее имя функции, арность, типы аргументов и возвращаемый тип данных. 
// В то время как определение функции описывает, что именно делает функция, прототип 
// функции может восприниматься как описание её интерфейса.

// Прототипы позволяют обращаться к функции до ее объявления
// Также это повышает удобство при работе с большим количеством функций, или при работе с вложенными функциями

void foo(); //Прототип
int foo2(int, int);

int main()
{
	foo();
	cout << foo2(5, 10);
	_getch();
    return 0;
}

void foo() {
	cout << "im a function" << endl;
}

int foo2(int a, int b) {
	return a + b;
}
