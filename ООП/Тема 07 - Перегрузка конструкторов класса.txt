/* Тема 6 - Перегрузка конструкторов класса */

#include <iostream>

using namespace std;

// С помощью перегрузки конструктора класса мы можем инициализировать клас любыми способами. 
// Под каждую ситуацию мы можем написать свой способ инициализации

// При перегрузке конструкторов они отличаются принимаемми параметрами и реализацией
class Point{
private:
    int x;
    int y;
public:
// Реализуем конструктор, который не принимает параметры
// Коструктор без принимаемых параметров - это конструктор по умолчанию, которому мы переопределяем реализацию
Point(){
    x = 0;
    y = 0;
}
// Реализуем конструктор, который принимает 2 параметра
Point(int valueX, int valueY){
    x = valueX;
    y = valueY;
};
// Реализуем еще 1 коструктор, который принимает 2 параметр
Point(int valueX){
    x = valueX;
    y = 0;
};
    void Print(){
        cout << x << " " << y << endl;
    }
};

int main()
{
    Point a(5, 10); 
    a.Print();
    Point b;
    b.Print();
    Point c(10);
    c.Print();
    return 0;
}
