// Геттеры, сеттеры

#include <iostream>

using namespace std;

class Point{
private:
    int x;
    int y; 
// Геттеры и сеттеры должны быть public - методами
public:
// Геттер
    int GetX(){
        return x;
    }
// Сеттер 
    void SetX(int value){
        x = value;
    }
// Функция для вывода
    void Print(){
        cout << x << " " << y << endl;
    }
};
    // Предоставлять доступ к важным свойствам класса и делать их модификатором public небезопасно. 
    // Это нарушает принцип инкапсуляции
    // Доступ к полям рекомендуется осуществлять через геттеры и сеттеры. Это функции, которые отвечают за взаимодействие 
    // с полями класса.
int main()
{
    Point a;
    a.Print();
    a.SetX(10); // Устанавливаем значение через Сеттер
    a.Print();
    cout << a.GetX(); // Получаем значение через Геттер
    // Аналогично геттеры и сеттеры реализуются для других полей класса
    // Это является примером инкапсудяции (сокрытия полей класса извне)
    return 0;
}
