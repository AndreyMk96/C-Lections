/* Тема 4 - Конструкторы класса. Конструктор по умолчанию */

#include <iostream>

using namespace std;

//Конструктор - это функци, которая вызывается автоматически, при создании объекта класса.
//Конструктор может принимать параметры, но не может возврщать занчения

class Point{
private:
    int x;
    int y;
public:
//У класса всегда есть конструктор, в случае если мы его не описали, компилятор добавляет его неявно,
// (он называется конструктором по умолчанию). Конструктор по умолчанию пустой.
// Конструктор называется также как и сам класс

//Конструктор должен быть в public секции, тк он должен быть доступен извне
Point(int valueX, int valueY){
    x = valueX;
    y = valueY;
    }
    
    void Print(){
        cout << x << " " << y << endl;
    }
};

int main()
{
    Point a(5, 10); // Теперь в объект класса надо передавать параметры
    a.Print();
    // Мы инициализируем класс в то состояние, в котором он должен хранить объекты
    // Конструктор нельзя вызвать через обьект класса. Только при создании.
    return 0;
}
